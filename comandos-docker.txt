docker ps
# Rodar container PostGres na versão 16.3 (modo interativo)
docker run --name librarydb -p 5432:5432 -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library -d postgres:16.3

# Para rodar o container em background
docker run --name librarydb -p 5432:5432 -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library postgres:16.3

# Rodar Pgadmin 4
docker run --name pgadmin4 -p 15432:80 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGAMDIN_DEFAULT_PASSWORD=admin dpage/pgadmin4

# Para rodar o container em background
docker run --name pgadmin4 -p 15432:80 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGAMDIN_DEFAULT_PASSWORD=admin -d dpage/pgadmin4

# Para a execução do container
docker stop pgadmin4
docker stop librarydb

 # Para excluir
 docker container rm pgadmin4

# Para vê containers ativos
docker ps
docker ps -a

# Para criar uma network
docker network create library-network


# Criando o container com a network
docker run --name librarydb -p 5432:5432 -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library --network library-network postgres:16.3

docker run --name pgadmin4 -p 15432:80 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin --network library-network dpage/pgadmin4

# Para checar a porta/serciço está disponivel
netstat -aof

# Para procurar porta especifica
netstat -aof | findstr 5432
